```javascript
describe('login 云函数', () => {
  it('应该成功登录老用户', async () => {
    const result = await cloud.callFunction({
      name: 'login',
      data: { code: 'test_code' }
    });
    expect(result.success).toBe(true);
  });
  
  it('应该成功注册新用户', async () => {
    const result = await cloud.callFunction({
      name: 'login',
      data: { 
        code: 'test_code',
        userInfo: { nickName: '新用户' }
      }
    });
    expect(result.success).toBe(true);
    expect(result.data.isNewUser).toBe(true);
  });
});
```
